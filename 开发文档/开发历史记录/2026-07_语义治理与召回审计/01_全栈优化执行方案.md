## 一、搜索性能优化（3.5秒 -> <500ms）

**预计收益**：用户感知搜索延迟从 3.5 秒降至 300-500ms，总对话等待从 7-12 秒降至 4-8 秒
**执行顺序**：行动1 → 行动2 → 行动3（互相独立可并行，但建议按顺序验证）
**风险点**：索引创建期间写性能略降（CONCURRENTLY 不锁表）；缓存需考虑内存占用
**回退方案**：DROP INDEX 即可回退索引；缓存代码保留原 DB 查询路径作 fallback

修改后：
```python
# 模块级缓存：owner_id -> (timestamp, [(normalized, term)])
_TERM_CACHE: dict[int, tuple[float, list[tuple[str, str]]]] = {}
_TERM_CACHE_TTL = 300  # 5分钟过期

async def _load_query_tokenizer_terms(db: AsyncSession | None, owner_id: int | None, query: str) -> list[str]:
    if db is None or owner_id is None or not hasattr(db, "execute"):
        return []
    compact = _normalize_term_text(query)
    if len(compact) < 2:
        return []

    import time as _time

    # 从缓存加载术语字典
    now = _time.time()
    cached = _TERM_CACHE.get(owner_id)
    if cached is None or (now - cached[0]) > _TERM_CACHE_TTL:
        try:
            from ..models import KbTerm

            result = await db.execute(
                select(KbTerm.normalized, KbTerm.term)
                .where(
                    KbTerm.owner_id == owner_id,
                    KbTerm.status == "active",
                    KbTerm.normalized != "",
                )
                .order_by(sa_text("length(kb_terms.normalized) DESC"), KbTerm.confidence.desc().nullslast())
            )
            pairs = [(str(row[0]), str(row[1])) for row in result.all() if row[0] and row[1]]
            _TERM_CACHE[owner_id] = (now, pairs)
        except Exception as exc:
            logger.warning("Query tokenizer terms cache load failed: %s", exc)
            return []
    else:
        pairs = cached[1]

    # 应用层子串匹配（Python 侧 in 操作，比 DB ILIKE 快 100 倍）
    matches = []
    for normalized, term in _TERM_CACHE[owner_id][1]:
        if normalized in compact:
            matches.append(term)
            if len(matches) >= 40:
                break
    return matches
```

```python
def _build_profile_input(fusions: list, all_text: str, max_chars: int = 12000) -> str:
    """优先用 page_summary 拼接（~120字/页），不截断全文。

    20 页文档 = 2400 字，远低于 12000 上限。
    只有当 page_summary 总字数不足 500 时才回退到 all_text 截断。
    """
    summaries = []
    for i, f in enumerate(fusions):
        data = f if isinstance(f, dict) else (getattr(f, "fused_data", None) or {})
        summary = data.get("page_summary") or data.get("fused_text", "")[:150]
        if summary.strip():
            summaries.append(f"[第{i+1}页] {summary.strip()}")
    summary_text = "\n".join(summaries)
    if len(summary_text) >= 500:
        return summary_text[:max_chars]
    # 回退：page_summary 不足时用原始全文截断
    return all_text[:max_chars]
```

```python
    *,
```

```python
    vector_store: str,
    fallback_from: str | None,
    allow: bool,
) -> bool:
    """当主 embedding 服务不可用时，是否降级到旧向量。

    配置项 KNOWLEDGE_VECTOR_FALLBACK 控制：
    - "disabled"（默认）: 永不降级

    import os

    if not allow:
        return False
```

```python
def _query_has_complex_intent(query: str, custom_terms: list[str] | None = None) -> bool:
    """检测查询是否包含复杂检索意图（需要 LLM 规划）。"""
    # 包含自定义术语命中 → 复杂
    if custom_terms and len(custom_terms) >= 2:
        return True
    # 包含文档类型关键词 → 复杂
    doc_type_keywords = {"报告", "方案", "标准", "规范", "合同", "配方", "检测", "研究", "临床", "实验"}
    if any(kw in query for kw in doc_type_keywords):
        return True
    # 包含多个实体（用中文标点/空格分割后有3+有意义段）
    segments = [s for s in re.split(r"[，。、\s]+", query) if len(s) >= 2]
    if len(segments) >= 3:
        return True
    return False
```
